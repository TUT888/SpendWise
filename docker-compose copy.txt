version: '3'
services:
  mongo:
    image: mongo
    container_name: mongo
    ports:
      - "27017:27017"
    volumes:
      - mongo-data:/data/db
    networks:
      - spendwise-network

  account-service:
    build: ./account_service
    container_name: account-service
    ports:
      - "3030:3030"
    environment:
      - MONGO_URL=mongodb://mongo:27017/spendwise
      - FRONTEND_ORIGIN=http://localhost:3080
      - SESSION_SECRET=MySecret
    depends_on:
      - mongo
    networks:
      - spendwise-network

  category-service:
    build: ./category_service
    container_name: category-service
    ports:
      - "3031:3031"
    environment:
      - MONGO_URL=mongodb://mongo:27017/spendwise
      - ACCOUNT_SERVICE_URL=http://account-service:3030
      - FRONTEND_ORIGIN=http://localhost:3080
    depends_on:
      - account-service
      - mongo
    networks:
      - spendwise-network

  expense-service:
    build: ./expense_service
    container_name: expense-service
    ports:
      - "3032:3032"
    environment:
      - MONGO_URL=mongodb://mongo:27017/spendwise
      - ACCOUNT_SERVICE_URL=http://account-service:3030
      - FRONTEND_ORIGIN=http://localhost:3080
    depends_on:
      - account-service
      - mongo
    networks:
      - spendwise-network

  frontend:
    build: ./frontend
    container_name: frontend
    ports:
      - "3080:3080"
    depends_on:
      - account-service
      - category-service
      - expense-service
    networks:
      - spendwise-network
    environment:
      - ACCOUNT_SERVICE_URL=http://account-service:3030
      - CATEGORY_SERVICE_URL=http://category-service:3031
      - EXPENSE_SERVICE_URL=http://expense-service:3032

volumes:
  mongo-data:

networks:
  spendwise-network:
    driver: bridge
